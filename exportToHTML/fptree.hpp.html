18   ForAll(Item) using Transaction           = std::vector<Item>;
19   ForAll(Item) using TransformedPrefixPath = std::pair<std::vector<Item>, uint64_t>;
20   ForAll(Item) using Pattern               = std::pair<std::set<Item>, uint64_t>;

22   ForAll(Item) struct FPNode {
23       const Item                           item;
24       uint64_t                             frequency;
25       std::shared_ptr<FPNode<Item>>        node_link;
26       std::weak_ptr<FPNode<Item>>          parent;
27       std::vector<std::shared_ptr<FPNode>> children;
28   
29       FPNode(const Item& item, const std::shared_ptr<FPNode<Item>>& parent) :
30               item( item ), frequency( 1 ), node_link( nullptr ), parent( parent ), children() {
31       }
32   };

34   ForAll(Item) struct FPTree {
35       std::shared_ptr<FPNode<Item>>                 root;
36       std::map<Item, std::shared_ptr<FPNode<Item>>> header_table;
37       uint64_t                                      minimum_support_threshold;
38   
39       // order items by decreasing frequency
40       struct frequency_comparator
41       {
42           bool operator()(const std::pair<Item, uint64_t> &lhs, const std::pair<Item, uint64_t> &rhs) const
43           {
44               return std::tie(lhs.second, lhs.first) > std::tie(rhs.second, rhs.first);
45           }
46       };
47   
48       FPTree(const std::vector<Transaction<Item>>& transactions, uint64_t minimum_support_threshold) :
49               root(std::make_shared<FPNode<Item>>( Item{}, nullptr )), header_table(),
50               minimum_support_threshold( minimum_support_threshold ) {
51   
52           std::cout << "initialization..." << std::endl;
53   
54           // scan the transactions counting the frequency of each item
55           std::map<Item, uint64_t> frequency_by_item;
56           for ( const Transaction<Item>& transaction : transactions ) {
57               for ( const Item& item : transaction ) {
58                   ++frequency_by_item[item];
59               }
60           }
61           // keep only items which have a frequency greater or equal than the minimum support threshold
62           for ( auto it = frequency_by_item.cbegin(); it != frequency_by_item.cend(); ) {
63               const uint64_t item_frequency = (*it).second;
64               if ( item_frequency < minimum_support_threshold ) { frequency_by_item.erase( it++ ); }
65               else { ++it; }
66           }
67   
68           std::set<std::pair<Item, uint64_t>, frequency_comparator> items_ordered_by_frequency(frequency_by_item.cbegin(), frequency_by_item.cend());
69   
70           // start tree construction
71           std::cout << "starting to construct the tree." << std::endl;
72   
73           // scan the transactions again
74           size_t i = 0;
75           for ( const Transaction<Item>& transaction : transactions ) {
76               auto curr_fpnode = root;
77               if (!(i % 1000)) std::cout << "Transaction #"<< i << std::endl;
78               i++;
79   
80               // select and sort the frequent items in transaction according to the order of items_ordered_by_frequency
81   
82               for ( const auto& pair : items_ordered_by_frequency ) {
83                   const Item& item = pair.first;
84   
85                   // check if item is contained in the current transaction
86                   if ( std::find( transaction.cbegin(), transaction.cend(), item ) != transaction.cend() ) {
87                       // insert item in the tree
88   
89                       // check if curr_fpnode has a child curr_fpnode_child such that curr_fpnode_child.item = item
90                       const auto it = std::find_if(
91                               curr_fpnode->children.cbegin(), curr_fpnode->children.cend(),  [item](const std::shared_ptr<FPNode<Item>>& fpnode) {
92                                   return fpnode->item == item;
93                               } );
94                       if ( it == curr_fpnode->children.cend() ) {
95                           // the child doesn't exist, create a new node
96                           const auto curr_fpnode_new_child = std::make_shared<FPNode<Item>>( item, curr_fpnode );
97   
98                           // add the new node to the tree
99                           curr_fpnode->children.push_back( curr_fpnode_new_child );
100  
101                          // update the node-link structure
102                          if ( header_table.count( curr_fpnode_new_child->item ) ) {
103                              auto prev_fpnode = header_table[curr_fpnode_new_child->item];
104                              while ( prev_fpnode->node_link ) { prev_fpnode = prev_fpnode->node_link; }
105                              prev_fpnode->node_link = curr_fpnode_new_child;
106                          }
107                          else {
108                              header_table[curr_fpnode_new_child->item] = curr_fpnode_new_child;
109                          }
110  
111                          // advance to the next node of the current transaction
112                          curr_fpnode = curr_fpnode_new_child;
113                      }
114                      else {
115                          // the child exist, increment its frequency
116                          auto curr_fpnode_child = *it;
117                          ++curr_fpnode_child->frequency;
118  
119                          // advance to the next node of the current transaction
120                          curr_fpnode = curr_fpnode_child;
121                      }
122                  }
123              }
124          }
125      }
126  
127      bool empty() const {
128          assert( root );
129          return root->children.size() == 0;
130      }
131  };

133  ForAll(Item) bool contains_single_path(const std::shared_ptr<FPNode<Item>>& fpnode) {
134      assert( fpnode );
135      if ( fpnode->children.size() == 0 ) { return true; }
136      if ( fpnode->children.size() > 1 ) { return false; }
137      return contains_single_path( fpnode->children.front() );
138  }
139  
140  ForAll(Item) bool contains_single_path(const FPTree<Item>& fptree) {
141      return fptree.empty() || contains_single_path( fptree.root );
142  }

144  ForAll(Item) std::set<Pattern<Item>> fptree_growth(const FPTree<Item>& fptree) {
145      if ( fptree.empty() ) { return {}; }
146  
147      if ( contains_single_path( fptree ) ) {
148          // generate all possible combinations of the items in the tree
149  
150          std::set<Pattern<Item>> single_path_patterns;
151  
152          // for each node in the tree
153          assert( fptree.root->children.size() == 1 );
154          auto curr_fpnode = fptree.root->children.front();
155          while ( curr_fpnode ) {
156              const Item& curr_fpnode_item = curr_fpnode->item;
157              const uint64_t curr_fpnode_frequency = curr_fpnode->frequency;
158  
159              // add a pattern formed only by the item of the current node
160              Pattern<Item> new_pattern{ { curr_fpnode_item }, curr_fpnode_frequency };
161              single_path_patterns.insert( new_pattern );
162  
163              // create a new pattern by adding the item of the current node to each pattern generated until now
164              for ( const Pattern<Item>& pattern : single_path_patterns ) {
165                  Pattern<Item> new_pattern{ pattern };
166                  new_pattern.first.insert( curr_fpnode_item );
167                  assert( curr_fpnode_frequency <= pattern.second );
168                  new_pattern.second = curr_fpnode_frequency;
169  
170                  single_path_patterns.insert( new_pattern );
171              }
172  
173              // advance to the next node until the end of the tree
174              assert( curr_fpnode->children.size() <= 1 );
175              if ( curr_fpnode->children.size() == 1 ) { curr_fpnode = curr_fpnode->children.front(); }
176              else { curr_fpnode = nullptr; }
177          }
178  
179          return single_path_patterns;
180      }
181      else {
182          // generate conditional fptrees for each different item in the fptree, then join the results
183  
184          std::set<Pattern<Item>> multi_path_patterns;
185  
186          // for each item in the fptree
187          for ( const auto& pair : fptree.header_table ) {
188              const Item& curr_item = pair.first;
189  
190              // build the conditional fptree relative to the current item
191  
192              // start by generating the conditional pattern base
193              std::vector<TransformedPrefixPath<Item>> conditional_pattern_base;
194  
195              // for each path in the header_table (relative to the current item)
196              auto path_starting_fpnode = pair.second;
197              while ( path_starting_fpnode ) {
198                  // construct the transformed prefix path
199  
200                  // each item in th transformed prefix path has the same frequency (the frequency of path_starting_fpnode)
201                  const uint64_t path_starting_fpnode_frequency = path_starting_fpnode->frequency;
202  
203                  auto curr_path_fpnode = path_starting_fpnode->parent.lock();
204                  // check if curr_path_fpnode is already the root of the fptree
205                  if ( curr_path_fpnode->parent.lock() ) {
206                      // the path has at least one node (excluding the starting node and the root)
207                      TransformedPrefixPath<Item> transformed_prefix_path{ {}, path_starting_fpnode_frequency };
208  
209                      while ( curr_path_fpnode->parent.lock() ) {
210                          assert( curr_path_fpnode->frequency >= path_starting_fpnode_frequency );
211                          transformed_prefix_path.first.push_back( curr_path_fpnode->item );
212  
213                          // advance to the next node in the path
214                          curr_path_fpnode = curr_path_fpnode->parent.lock();
215                      }
216  
217                      conditional_pattern_base.push_back( transformed_prefix_path );
218                  }
219  
220                  // advance to the next path
221                  path_starting_fpnode = path_starting_fpnode->node_link;
222              }
223  
224              // generate the transactions that represent the conditional pattern base
225              std::vector<Transaction<Item>> conditional_fptree_transactions;
226              for ( const TransformedPrefixPath<Item>& transformed_prefix_path : conditional_pattern_base ) {
227                  const std::vector<Item>& transformed_prefix_path_items = transformed_prefix_path.first;
228                  const uint64_t transformed_prefix_path_items_frequency = transformed_prefix_path.second;
229  
230                  Transaction<Item> transaction = transformed_prefix_path_items;
231  
232                  // add the same transaction transformed_prefix_path_items_frequency times
233                  for ( auto i = 0; i < transformed_prefix_path_items_frequency; ++i ) {
234                      conditional_fptree_transactions.push_back( transaction );
235                  }
236              }
237  
238              // build the conditional fptree relative to the current item with the transactions just generated
239              const FPTree<Item> conditional_fptree( conditional_fptree_transactions, fptree.minimum_support_threshold );
240              // call recursively fptree_growth on the conditional fptree (empty fptree: no patterns)
241              std::set<Pattern<Item>> conditional_patterns = fptree_growth( conditional_fptree );
242  
243              // construct patterns relative to the current item using both the current item and the conditional patterns
244              std::set<Pattern<Item>> curr_item_patterns;
245  
246              // the first pattern is made only by the current item
247              // compute the frequency of this pattern by summing the frequency of the nodes which have the same item (follow the node links)
248              uint64_t curr_item_frequency = 0;
249              auto fpnode = pair.second;
250              while ( fpnode ) {
251                  curr_item_frequency += fpnode->frequency;
252                  fpnode = fpnode->node_link;
253              }
254              // add the pattern as a result
255              Pattern<Item> pattern{ { curr_item }, curr_item_frequency };
256              curr_item_patterns.insert( pattern );
257  
258              // the next patterns are generated by adding the current item to each conditional pattern
259              for ( const Pattern<Item>& pattern : conditional_patterns ) {
260                  Pattern<Item> new_pattern{ pattern };
261                  new_pattern.first.insert( curr_item );
262                  assert( curr_item_frequency >= pattern.second );
263                  new_pattern.second = pattern.second;
264  
265                  curr_item_patterns.insert( { new_pattern } );
266              }
267  
268              // join the patterns generated by the current item with all the other items of the fptree
269              multi_path_patterns.insert( curr_item_patterns.cbegin(), curr_item_patterns.cend() );
270          }
271  
272          return multi_path_patterns;
273      }
274  }
